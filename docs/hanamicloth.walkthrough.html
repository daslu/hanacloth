2.8 Example: out-of-sample predictions

(-> (toydata/iris-ds)
    (tc/concat (tc/dataset {:sepal_width (range 4 10)
                            :sepal_length (repeat 6 nil)}))
    (tc/map-columns :relative-time
                    [:sepal_length]
                    #(if % "Past" "Future"))
    (hanami/base #:hanami{:x :sepal_width
                          :y :sepal_length
                          :color "relative-time"
                          :group []})
    hanami/layer-point
    hanami/layer-smooth)

(-> (toydata/iris-is) (hanami/plot ht/rule-chart {:X :sepal-width :Y :sepal-length :X2 :petal-width :Y2 :petal-length :OPACITY 0.2 :SIZE 3 :COLOR “species”}))
